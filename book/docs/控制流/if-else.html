if-else

执行下面的代码，进行条件判断。

#include <stdio.h>

int main(int argc, char const *argv[])
{
    int a = 233;
    if (a > 0)
    {
        printf("Yes.\n");
    }
    else
    {
        printf("No.\n");
    }
    return 0;
}
