多维数组

多维数组

二维数组实际上是一种特殊的一维数组，它的每个元素也是一个一维数组。

#include <stdio.h>

int main(int argc, char const *argv[])
{
    char day[12][4] = {{"Jan"}, {"Feb"}, {"Mar"}, {"Apr"}, {"May"}, {"Jun"}, {"Jul"}, {"Aug"}, {"Sep"}, {"Oct"}, {"Nov"}, {"Dec"}};
    printf("%s\n", day[0]);
    printf("%s\n", day[6]);
    printf("%s\n", day[11]);
    return 0;
}
